REGISTER IN WAVR PRO
// WavrPro.jsx — add these
import DrumMachine
  from "./plugins/DrumMachine";

// Inside component:
const pm = usePluginManager({
  audioCtxRef, bpmRef ...
});

useEffect(() => {
  pm.register(DrumMachine);
}, []);

// In topbar:
<button onClick={() =>
  pm.toggle("drum-machine")}>
  Drums
</button>

// In JSX:
{pm.plugins.filter(p => p.isOpen)
  .map(p => (
  <PluginShell
    pluginClass={p.cls}
    hostAPI={pm.getHostAPI(p.id)}
    open={p.isOpen}
    onClose={() =>
      pm.close(p.id)}
  />
))}
